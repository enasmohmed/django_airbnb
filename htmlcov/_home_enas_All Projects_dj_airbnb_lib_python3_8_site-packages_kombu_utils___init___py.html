48def symbol_by_name(name, aliases={}, imp=None, package=None, 

49 sep='.', default=None, **kwargs): 

50 """Get symbol by qualified name. 

51 

52 The name should be the full dot-separated path to the class:: 

53 

54 modulename.ClassName 

55 

56 Example:: 

57 

58 celery.concurrency.processes.TaskPool 

59 ^- class name 

60 

61 or using ':' to separate module and symbol:: 

62 

63 celery.concurrency.processes:TaskPool 

64 

65 If `aliases` is provided, a dict containing short name/long name 

66 mappings, the name is looked up in the aliases first. 

67 

68 Examples: 

69 

70 >>> symbol_by_name('celery.concurrency.processes.TaskPool') 

71 <class 'celery.concurrency.processes.TaskPool'> 

72 

73 >>> symbol_by_name('default', { 

74 ... 'default': 'celery.concurrency.processes.TaskPool'}) 

75 <class 'celery.concurrency.processes.TaskPool'> 

76 

77 # Does not try to look up non-string names. 

78 >>> from celery.concurrency.processes import TaskPool 

79 >>> symbol_by_name(TaskPool) is TaskPool 

80 True 

81 

82 """ 

83 if imp is None: 

84 imp = importlib.import_module 

85 

86 if not isinstance(name, string_t): 

87 return name # already a class 

88 

89 name = aliases.get(name) or name 

90 sep = ':' if ':' in name else sep 

91 module_name, _, cls_name = name.rpartition(sep) 

92 if not module_name: 

93 cls_name, module_name = None, package if package else cls_name 

94 try: 

95 try: 

96 module = imp(module_name, package=package, **kwargs) 

97 except ValueError as exc: 

98 reraise(ValueError, 

99 ValueError("Couldn't import {0!r}: {1}".format(name, exc)), 

100 sys.exc_info()[2]) 

101 return getattr(module, cls_name) if cls_name else module 

102 except (ImportError, AttributeError): 

103 if default is None: 

104 raise 

105 return default

189def fxrange(start=1.0, stop=None, step=1.0, repeatlast=False): 

190 cur = start * 1.0 

191 while 1: 

192 if not stop or cur <= stop: 

193 yield cur 

194 cur += step 

195 else: 

196 if not repeatlast: 

197 break 

198 yield cur - step

201def fxrangemax(start=1.0, stop=None, step=1.0, max=100.0): 

202 sum_, cur = 0, start * 1.0 

203 while 1: 

204 if sum_ >= max: 

205 break 

206 yield cur 

207 if stop: 

208 cur = min(cur + step, stop) 

209 else: 

210 cur += step 

211 sum_ += cur

214def retry_over_time(fun, catch, args=[], kwargs={}, errback=None, 

215 max_retries=None, interval_start=2, interval_step=2, 

216 interval_max=30, callback=None): 

217 """Retry the function over and over until max retries is exceeded. 

218 

219 For each retry we sleep a for a while before we try again, this interval 

220 is increased for every retry until the max seconds is reached. 

221 

222 :param fun: The function to try 

223 :param catch: Exceptions to catch, can be either tuple or a single 

224 exception class. 

225 :keyword args: Positional arguments passed on to the function. 

226 :keyword kwargs: Keyword arguments passed on to the function. 

227 :keyword errback: Callback for when an exception in ``catch`` is raised. 

228 The callback must take two arguments: ``exc`` and ``interval``, where 

229 ``exc`` is the exception instance, and ``interval`` is the time in 

230 seconds to sleep next.. 

231 :keyword max_retries: Maximum number of retries before we give up. 

232 If this is not set, we will retry forever. 

233 :keyword interval_start: How long (in seconds) we start sleeping between 

234 retries. 

235 :keyword interval_step: By how much the interval is increased for each 

236 retry. 

237 :keyword interval_max: Maximum number of seconds to sleep between retries. 

238 

239 """ 

240 retries = 0 

241 interval_range = fxrange(interval_start, 

242 interval_max + interval_start, 

243 interval_step, repeatlast=True) 

244 for retries in count(): 

245 try: 

246 return fun(*args, **kwargs) 

247 except catch as exc: 

248 if max_retries and retries >= max_retries: 

249 raise 

250 if callback: 

251 callback() 

252 tts = float(errback(exc, interval_range, retries) if errback 

253 else next(interval_range)) 

254 if tts: 

255 for _ in range(int(tts)): 

256 if callback: 

257 callback() 

258 sleep(1.0) 

259 # sleep remainder after int truncation above. 

260 sleep(abs(int(tts) - tts))

365@contextmanager 

366def nested(*managers): # pragma: no cover 

367 # flake8: noqa 

368 """Combine multiple context managers into a single nested 

369 context manager.""" 

370 exits = [] 

371 vars = [] 

372 exc = (None, None, None) 

373 try: 

374 try: 

375 for mgr in managers: 

376 exit = mgr.__exit__ 

377 enter = mgr.__enter__ 

378 vars.append(enter()) 

379 exits.append(exit) 

380 yield vars 

381 except: 

382 exc = sys.exc_info() 

383 finally: 

384 while exits: 

385 exit = exits.pop() 

386 try: 

387 if exit(*exc): 

388 exc = (None, None, None) 

389 except: 

390 exc = sys.exc_info() 

391 if exc != (None, None, None): 

392 # Don't rely on sys.exc_info() still containing 

393 # the right information. Another exception may 

394 # have been raised and caught by an exit method 

395 reraise(exc[0], exc[1], exc[2]) 

396 finally: 

397 del(exc)
