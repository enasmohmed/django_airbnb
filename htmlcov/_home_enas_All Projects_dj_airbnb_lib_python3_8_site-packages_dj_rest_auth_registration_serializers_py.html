98 access_token = attrs.get('access_token') 

99 code = attrs.get('code') 

100 # Case 1: We received the access_token

102 tokens_to_parse = {'access_token': access_token} 

103 token = access_token 

104 # For sign in with apple 

105 id_token = attrs.get('id_token') 

106 if id_token: 

107 tokens_to_parse['id_token'] = id_token 

108 

109 # Case 2: We received the authorization code 

110 elif code: 

111 self.set_callback_url(view=view, adapter_class=adapter_class) 

112 self.client_class = getattr(view, 'client_class', None)

121 client = self.client_class(

124 app.secret, 

125 adapter.access_token_method, 

126 adapter.access_token_url, 

127 self.callback_url, 

128 scope, 

129 scope_delimiter=adapter.scope_delimiter, 

130 headers=adapter.headers, 

131 basic_auth=adapter.basic_auth,
