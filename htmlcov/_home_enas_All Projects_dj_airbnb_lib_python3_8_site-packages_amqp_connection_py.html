219 def _wait_method(self, channel_id, allowed_methods, timeout=None): 

220 """Wait for a method from the server destined for

223 # Check the channel's deferred methods 

224 #

227 for queued_method in method_queue: 

228 method_sig = queued_method[0] 

229 if (allowed_methods is None) \ 

230 or (method_sig in allowed_methods) \ 

231 or (method_sig == (20, 40)):

237 #

400 """Request a connection close 

401 

402 This method indicates that the sender wants to close the

404 forced shut-down) or due to an error handling a specific

406 exception, the sender provides the class and method id of the 

407 method which caused the exception. 

408 

409 RULE: 

410 

411 After sending this method any received method except the 

412 Close-OK method MUST be discarded. 

413 

414 RULE: 

415

417 to detect failure of the other peer to respond correctly 

418 with the Close-OK method. 

419

774 which must be one of those specified by the server. 

775 

776 RULE: 

777 

778 The client SHOULD authenticate using the highest- 

779 level security profile it can handle from the list 

780 provided by the server. 

781 

782 RULE: 

783 

784 The mechanism field MUST contain one of the 

785 security mechanisms proposed by the server in the 

786 Start method. If it doesn't, the server MUST close 

787 the socket. 

788 

789 response: longstr 

790 

791 security response data 

792 

793 A block of opaque data passed to the security 

794 mechanism. The contents of this data are defined by 

795 the SASL security mechanism. For the PLAIN security 

796 mechanism this is defined as a field table holding two 

797 fields, LOGIN and PASSWORD. 

798 

799 locale: shortstr 

800 

801 selected message locale 

802 

803 A single message local selected by the client, which 

804 must be one of those specified by the server.

806 """ 

807 if self.server_capabilities.get('consumer_cancel_notify'): 

808 if 'capabilities' not in client_properties: 

809 client_properties['capabilities'] = {} 

810 client_properties['capabilities']['consumer_cancel_notify'] = True 

811 if self.server_capabilities.get('connection.blocked'): 

812 if 'capabilities' not in client_properties:

822 def _tune(self, args): 

823 """Propose connection tuning parameters 

824 

825 This method proposes a set of connection configuration values 

826 to the client. The client can accept and/or adjust these. 

827 

828 PARAMETERS: 

829 channel_max: short 

830 

831 proposed maximum channels 

832 

833 The maximum total number of channels that the server 

834 allows per connection. Zero means that the server does 

835 not impose a fixed limit, but the number of allowed 

836 channels may be limited by available server resources. 

837 

838 frame_max: long 

839 

840 proposed maximum frame size 

841 

842 The largest frame size that the server proposes for 

843 the connection. The client can negotiate a lower 

844 value. Zero means that the server does not impose any 

845 specific limit but may reject very large frames if it 

846 cannot allocate resources for them. 

847 

848 RULE: 

849 

850 Until the frame-max has been negotiated, both 

851 peers MUST accept frames of up to 4096 octets 

852 large. The minimum non-zero value for the frame- 

853 max field is 4096. 

854 

855 heartbeat: short 

856 

857 desired heartbeat delay 

858 

859 The delay, in seconds, of the connection heartbeat 

860 that the server wants. Zero means the server does not 

861 want a heartbeat. 

862 

863 """ 

864 client_heartbeat = self.client_heartbeat or 0 

865 self.channel_max = args.read_short() or self.channel_max 

866 self.frame_max = args.read_long() or self.frame_max 

867 self.method_writer.frame_max = self.frame_max 

868 self.server_heartbeat = args.read_short() or 0 

869 

870 # negotiate the heartbeat interval to the smaller of the 

871 # specified values 

872 if self.server_heartbeat == 0 or client_heartbeat == 0: 

873 self.heartbeat = max(self.server_heartbeat, client_heartbeat) 

874 else: 

875 self.heartbeat = min(self.server_heartbeat, client_heartbeat) 

876 

877 # Ignore server heartbeat if client_heartbeat is disabled 

878 if not self.client_heartbeat: 

879 self.heartbeat = 0 

880 

881 self._x_tune_ok(self.channel_max, self.frame_max, self.heartbeat)

883 def send_heartbeat(self): 

884 self.transport.write_frame(8, 0, bytes())

886 def heartbeat_tick(self, rate=2): 

887 """Send heartbeat packets, if necessary, and fail if none have been 

888 received recently. This should be called frequently, on the order of 

889 once per second. 

890 

891 :keyword rate: Ignored 

892 """ 

893 if not self.heartbeat: 

894 return 

895 

896 # treat actual data exchange in either direction as a heartbeat 

897 sent_now = self.method_writer.bytes_sent 

898 recv_now = self.method_reader.bytes_recv 

899 if self.prev_sent is None or self.prev_sent != sent_now: 

900 self.last_heartbeat_sent = monotonic() 

901 if self.prev_recv is None or self.prev_recv != recv_now: 

902 self.last_heartbeat_received = monotonic() 

903 self.prev_sent, self.prev_recv = sent_now, recv_now 

904 

905 # send a heartbeat if it's time to do so 

906 if monotonic() > self.last_heartbeat_sent + self.heartbeat: 

907 self.send_heartbeat() 

908 self.last_heartbeat_sent = monotonic() 

909 

910 # if we've missed two intervals' heartbeats, fail; this gives the 

911 # server enough time to send heartbeats a little late 

912 if (self.last_heartbeat_received and 

913 self.last_heartbeat_received + 2 * 

914 self.heartbeat < monotonic()): 

915 raise ConnectionForced('Too many heartbeats missed')
