232def flatten(d, ns=''): 

233 stack = deque([(ns, d)]) 

234 while stack: 

235 name, space = stack.popleft() 

236 for key, value in items(space): 

237 if isinstance(value, dict): 

238 stack.append((name + key + '_', value)) 

239 else: 

240 yield name + key, value

244def find_deprecated_settings(source): 

245 from celery.utils import warn_deprecated 

246 for name, opt in flatten(NAMESPACES): 

247 if (opt.deprecate_by or opt.remove_by) and getattr(source, name, None): 

248 warn_deprecated(description='The {0!r} setting'.format(name), 

249 deprecation=opt.deprecate_by, 

250 removal=opt.remove_by, 

251 alternative='Use the {0.alt} instead'.format(opt)) 

252 return source

255@memoize(maxsize=None) 

256def find(name, namespace='celery'): 

257 # - Try specified namespace first. 

258 namespace = namespace.upper() 

259 try: 

260 return searchresult( 

261 namespace, name.upper(), NAMESPACES[namespace][name.upper()], 

262 ) 

263 except KeyError: 

264 # - Try all the other namespaces. 

265 for ns, keys in items(NAMESPACES): 

266 if ns.upper() == name.upper(): 

267 return searchresult(None, ns, keys) 

268 elif isinstance(keys, dict): 

269 try: 

270 return searchresult(ns, name.upper(), keys[name.upper()]) 

271 except KeyError: 

272 pass 

273 # - See if name is a qualname last. 

274 return searchresult(None, name.upper(), DEFAULTS[name.upper()])
