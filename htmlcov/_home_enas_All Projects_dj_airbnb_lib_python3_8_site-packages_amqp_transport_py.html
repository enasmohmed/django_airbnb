53class _AbstractTransport(object): 

54 """Common superclass for TCP and SSL transports""" 

55 connected = False 

56 

57 def __init__(self, host, connect_timeout): 

58 self.connected = True 

59 msg = None 

60 port = AMQP_PORT 

61 

62 m = IPV6_LITERAL.match(host) 

63 if m: 

64 host = m.group(1) 

65 if m.group(2): 

66 port = int(m.group(2)) 

67 else: 

68 if ':' in host: 

69 host, port = host.rsplit(':', 1) 

70 port = int(port) 

71 

72 self.sock = None 

73 last_err = None 

74 for res in socket.getaddrinfo(host, port, 0, 

75 socket.SOCK_STREAM, SOL_TCP): 

76 af, socktype, proto, canonname, sa = res 

77 try: 

78 self.sock = socket.socket(af, socktype, proto) 

79 try: 

80 set_cloexec(self.sock, True) 

81 except NotImplementedError: 

82 pass 

83 self.sock.settimeout(connect_timeout) 

84 self.sock.connect(sa) 

85 except socket.error as exc: 

86 msg = exc 

87 self.sock.close() 

88 self.sock = None 

89 last_err = msg 

90 continue 

91 break 

92 

93 if not self.sock: 

94 # Didn't connect, return the most recent error message 

95 raise socket.error(last_err) 

96 

97 try: 

98 self.sock.settimeout(None) 

99 self.sock.setsockopt(SOL_TCP, socket.TCP_NODELAY, 1) 

100 self.sock.setsockopt(socket.SOL_SOCKET, socket.SO_KEEPALIVE, 1) 

101 

102 self._setup_transport() 

103 

104 self._write(AMQP_PROTOCOL_HEADER) 

105 except (OSError, IOError, socket.error) as exc: 

106 if get_errno(exc) not in _UNAVAIL: 

107 self.connected = False 

108 raise

150 def read_frame(self, unpack=unpack): 

151 read = self._read 

152 read_frame_buffer = EMPTY_BUFFER 

153 try: 

154 frame_header = read(7, True) 

155 read_frame_buffer += frame_header 

156 frame_type, channel, size = unpack('>BHI', frame_header) 

157 payload = read(size) 

158 read_frame_buffer += payload 

159 ch = ord(read(1)) 

160 except socket.timeout: 

161 self._read_buffer = read_frame_buffer + self._read_buffer 

162 raise 

163 except (OSError, IOError, socket.error) as exc: 

164 # Don't disconnect for ssl read time outs 

165 # http://bugs.python.org/issue10272 

166 if isinstance(exc, SSLError) and 'timed out' in str(exc): 

167 raise socket.timeout() 

168 if get_errno(exc) not in _UNAVAIL: 

169 self.connected = False 

170 raise 

171 if ch == 206: # '\xce' 

172 return frame_type, channel, payload 

173 else: 

174 raise UnexpectedFrame( 

175 'Received 0x{0:02x} while expecting 0xce'.format(ch))

177 def write_frame(self, frame_type, channel, payload): 

178 size = len(payload) 

179 try: 

180 self._write(pack( 

181 '>BHI%dsB' % size, 

182 frame_type, channel, size, payload, 0xce, 

183 )) 

184 except socket.timeout: 

185 raise 

186 except (OSError, IOError, socket.error) as exc: 

187 if get_errno(exc) not in _UNAVAIL: 

188 self.connected = False 

189 raise

192class SSLTransport(_AbstractTransport): 

193 """Transport that works over SSL""" 

194 

195 def __init__(self, host, connect_timeout, ssl): 

196 if isinstance(ssl, dict): 

197 self.sslopts = ssl 

198 self._read_buffer = EMPTY_BUFFER 

199 super(SSLTransport, self).__init__(host, connect_timeout) 

200 

201 def _setup_transport(self): 

202 """Wrap the socket in an SSL object.""" 

203 if hasattr(self, 'sslopts'): 

204 self.sock = ssl.wrap_socket(self.sock, **self.sslopts) 

205 else: 

206 self.sock = ssl.wrap_socket(self.sock) 

207 self.sock.do_handshake() 

208 self._quick_recv = self.sock.read 

209 

210 def _shutdown_transport(self): 

211 """Unwrap a Python 2.6 SSL socket, so we can call shutdown()""" 

212 if self.sock is not None: 

213 try: 

214 unwrap = self.sock.unwrap 

215 except AttributeError: 

216 return 

217 self.sock = unwrap() 

218 

219 def _read(self, n, initial=False, 

220 _errnos=(errno.ENOENT, errno.EAGAIN, errno.EINTR)): 

221 # According to SSL_read(3), it can at most return 16kb of data. 

222 # Thus, we use an internal read buffer like TCPTransport._read 

223 # to get the exact number of bytes wanted. 

224 recv = self._quick_recv 

225 rbuf = self._read_buffer 

226 try: 

227 while len(rbuf) < n: 

228 try: 

229 s = recv(n - len(rbuf)) # see note above 

230 except socket.error as exc: 

231 # ssl.sock.read may cause ENOENT if the 

232 # operation couldn't be performed (Issue celery#1414). 

233 if not initial and exc.errno in _errnos: 

234 continue 

235 raise 

236 if not s: 

237 raise IOError('Socket closed') 

238 rbuf += s 

239 except: 

240 self._read_buffer = rbuf 

241 raise 

242 result, self._read_buffer = rbuf[:n], rbuf[n:] 

243 return result 

244 

245 def _write(self, s): 

246 """Write a string out to the SSL socket fully.""" 

247 try: 

248 write = self.sock.write 

249 except AttributeError: 

250 # Works around a bug in python socket library 

251 raise IOError('Socket closed') 

252 else: 

253 while s: 

254 n = write(s) 

255 if not n: 

256 raise IOError('Socket closed') 

257 s = s[n:]

260class TCPTransport(_AbstractTransport): 

261 """Transport that deals directly with TCP socket.""" 

262 

263 def _setup_transport(self): 

264 """Setup to _write() directly to the socket, and 

265 do our own buffered reads.""" 

266 self._write = self.sock.sendall 

267 self._read_buffer = EMPTY_BUFFER 

268 self._quick_recv = self.sock.recv 

269 

270 def _read(self, n, initial=False, _errnos=(errno.EAGAIN, errno.EINTR)): 

271 """Read exactly n bytes from the socket""" 

272 recv = self._quick_recv 

273 rbuf = self._read_buffer 

274 try: 

275 while len(rbuf) < n: 

276 try: 

277 s = recv(n - len(rbuf)) 

278 except socket.error as exc: 

279 if not initial and exc.errno in _errnos: 

280 continue 

281 raise 

282 if not s: 

283 raise IOError('Socket closed') 

284 rbuf += s 

285 except: 

286 self._read_buffer = rbuf 

287 raise 

288 

289 result, self._read_buffer = rbuf[:n], rbuf[n:] 

290 return result
